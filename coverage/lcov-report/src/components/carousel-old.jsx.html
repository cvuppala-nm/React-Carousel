import React, { useEffect, useState, useRef, useMemo } from 'react';
 
import '../styles/carousel.scss';
import { getChildren } from '../utils/helpers';
import { MAX_SLIDES, SLIDE_WIDTH } from '../constants/appConstants';
 
const Carousel = ({
  cards,
  slidesToShow,
  onAddCard,
  onCardDelete,
  onCardUpdate,
  onUpdateSlidesToShow
}) => {
  // slides parent component
  const slidesContainerRef = useRef(null);
 
  // total cards length
  const totalCards = useRef(cards.length);
 
  // current index of slides
  const [slideIndex, setSlideIndex] = useState(1);
  
  // total slides
  const [totalNodes, setTotalNodes] = useState(cards?.length ?? 0);
 
  // on mount
  useEffect(() => {
    init();
 
    // on unmount
    return () => {
      removeEvents();
    }
  }, []);
 
  useEffect(() => {
    if (totalCards.current < cards.length) {
      addMirrorNodes();
    } else {
      calculateTotalNodes();
    }
    
    totalCards.current = cards.length;
  }, [cards.length]);
 
  /** disable next and prev button */
  const disableControls = useMemo(() => cards.length < slidesToShow, [cards.length, slidesToShow]);
 
  /** calculate container width */
  const listStyles = useMemo(() => {
    // calculate the actual visible slides
    const slides = Math.min(totalNodes, slidesToShow);
    return {
      // todo find alternative way to fix the border issue
      // total slides - 2px  ** ( -2px to hide last slide border right)
      width: `calc(${slides * SLIDE_WIDTH}px - 2px)`,
    }
  }, [slidesToShow, totalNodes]);
 
  /** calculate track width */
  const trackStyles = useMemo(() => {
    // calculate offset for transformation
    const offset = (SLIDE_WIDTH * slideIndex) * -1;
    return {
      transform: `translate3d(${offset}px, 0px, 0px)`,
      // total nodes * slide width
      width: `calc(${(totalNodes) * SLIDE_WIDTH}px)`,
    }
  }, [totalNodes, slideIndex]); 
 
 
  /** initial component setup */
  const init = () => {
    addMirrorNodes();
 
    addEvents();
 
    calculateMaxSlides();
  };
 
  /** calculate max slides to display according to screen size */
  const calculateMaxSlides = () => {
    const n = Math.floor(window.innerWidth / (SLIDE_WIDTH + 100));
    const slides = Math.min(MAX_SLIDES, n);
    onUpdateSlidesToShow(slides);
  };
 
  /** handle window resize for responsiveness */
  const handleWindowResize = () => {
    calculateMaxSlides();
  };
 
  /** add event listeners */
  const addEvents = () => {
    window.addEventListener('resize', handleWindowResize);
  };
 
  /** remove event listeners */
  const removeEvents = () => {
    window.removeEventListener('resize', handleWindowResize);
  };
 
  /** this will mirror first and last node for infinite scroll */
  const addMirrorNodes = () => {
    const [fNode, lNode] = getFirstAndLastNode();
    
    if (fNode && lNode) {
      const el = slidesContainerRef.current;
 
      const fMirrorNode = fNode.cloneNode(true);
      const lMirrorNode = lNode.cloneNode(true);
 
      fMirrorNode.className = fMirrorNode.className.indexOf('clone-slide') === -1
        ? `${fMirrorNode.className} clone-slide`
        :  fMirrorNode.className;
 
      lMirrorNode.className = lMirrorNode.className.indexOf('clone-slide') === -1
        ? `${lMirrorNode.className} clone-slide`
        :  lMirrorNode.className;
 
      el.appendChild(fMirrorNode);
      el.insertBefore(lMirrorNode, el.firstChild);
 
      calculateTotalNodes();
    }
  };
 
  /** this will remove mirror slides */
  const removeMirrorNodes = () => {
    const [fNode, lNode] = getFirstAndLastNode();
    
    if (fNode && lNode) {
      const el = slidesContainerRef.current;
 
      el.removeChild(fNode);
      el.removeChild(lNode);
    }
  };
 
 
  /** check mirror nodes after deleting node */
  const checkMirrorNodesAfterDelete = (child, index) => {
    const el = slidesContainerRef.current;
 
    if (index === 0) {
      const s = child.nextElementSibling;
 
      el.removeChild(el.lastChild);
 
      const mirrorNode = s.cloneNode(true);
 
      mirrorNode.className = mirrorNode.className.indexOf('clone-slide') === -1
        ? `${mirrorNode.className} clone-slide`
        :  mirrorNode.className;
 
      el.appendChild(mirrorNode);
    } else if (index === (cards.length - 1)) {
      const s = child.previousElementSibling;
 
      el.removeChild(el.firstChild);
 
      const mirrorNode = s.cloneNode(true);
 
      mirrorNode.className = mirrorNode.className.indexOf('clone-slide') === -1
        ? `${mirrorNode.className} clone-slide`
        :  mirrorNode.className;
 
      el.insertBefore(mirrorNode, el.firstChild);
    }
  };
 
  /** get first and last node of slider */
  const getFirstAndLastNode = () => {
    const children = getChildren(slidesContainerRef.current);
 
    if (!children?.length) {
      return [null, null];
    }
 
    const firstChild = children[0];
    const lastChild = children[children.length - 1];
 
    return [firstChild, lastChild];
  }
 
  /** calculate total slides */
  const calculateTotalNodes = () => {
    const children = getChildren(slidesContainerRef.current);
 
    setTotalNodes(children?.length ?? 0)
  };
 
  /** handle next button click */
  const handleNextClick = () => {
    const nextIndex = slideIndex + 1;
    // if current slide is last slide then reset index to 0
    const currentIndex = nextIndex > (cards.length - 1)
      ? 0 : nextIndex;
    setSlideIndex(currentIndex);
  };
 
  /** handle prev button click  */
  const handlePrevClick = () => {
    const prevIndex = slideIndex - 1;
    // if current slide is first slide then set current index to last slide's index
    const currentIndex = slideIndex === 0
      ? (cards.length - 1) : prevIndex;
    setSlideIndex(currentIndex);
  };
 
  /** handle on add button click */
  const handleOnAddCard = () => {
    const i = cards?.length
      ? cards[cards.length - 1].index
      : -1;
 
    removeMirrorNodes();
    
    onAddCard(i + 1);
  };
 
  /** handle on card update */
  const handleOnCardUpdate = (e, index) => {
    const el = slidesContainerRef.current;
 
    if (index === 0) {
      el.lastChild.querySelector('h3').textContent = e.target.value;
    }
 
    if (index === (cards.length - 1)) {
      el.firstChild.querySelector('h3').textContent = e.target.value;
    }
    
    onCardUpdate(e, index);
  };
 
  /** handle on card delete */
  const handleOnCardDelete = (card, index) => {
    const el = slidesContainerRef.current;
 
    const child = el.querySelector(`[data-index='${card.index}']:not(.clone-slide)`)
    if (child) {
      checkMirrorNodesAfterDelete(child, index);
    }
 
    onCardDelete(index);
  };
 
  return (
    <>
      <div className="carousel-wrapper" role="toolbar">
        <button
          type="button"
          className="carousel-nav-btn"
          aria-label="Previous"
          onClick={handlePrevClick}
          disabled={disableControls}
        >
          <i className="arrow left"></i>
        </button>
        <div className="carousel-list" style={listStyles}>
          <div
            className="carousel-track"
            role="listbox"
            ref={slidesContainerRef}
            style={trackStyles}
          >
            {cards.map((card) => (
              <div
                key={card.id}
                className="slide"
                data-index={card.index}
                aria-hidden="true"
                tabIndex="-1"
              >
                <h3>{card.text}</h3>
              </div>
            ))}
          </div>
        </div>
        <button 
          type="button" 
          className="carousel-nav-btn" 
          aria-label="Next"
          onClick={handleNextClick}
          disabled={disableControls}
        >
          <i className="arrow right"></i>
        </button>
      </div>
      <hr />
      <div className="slides-heading">
        <h3>Slides - ({cards.length})</h3>
 
        <button type="button" onClick={handleOnAddCard}>
          Add Slide
        </button>
      </div>
      <div className="slides-inputs">
        {cards.map((card, index) => (
          <div key={card.id}>
            <input
              type="text"
              value={card.text}
              onChange={(e) => handleOnCardUpdate(e, index)}
            />
            {(cards.length > 1) && (
              <button
                type="button"
                onClick={() => handleOnCardDelete(card, index)}>
                delete
              </button>
            )}
          </div>
        ))}
      </div>
    </>
  )
};
 
export default Carousel;
